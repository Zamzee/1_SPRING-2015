using System;

class Program
{
    const int SEARCH_TERM = 12;

    static void Main()
    {
        int[ ] values = {1, 3, 7, 12, 15};
        int foundAt = -1;
        int beginIndex = 0;
        int endIndex = values.Length-1;

        // Search the array to find the value
        // This will be a binary search
        // return the index of the element, or -1 if not found
        while(foundAt == -1 && beginIndex <= endIndex)
        {
            int midPoint = (beginIndex + endIndex) / 2;
            
            if (SEARCH_TERM < values[midPoint])
                endIndex = midPoint - 1;
            else if (SEARCH_TERM > values[midPoint])
                beginIndex = midPoint + 1;  
            else
                foundAt = midPoint;
        }
        
        if (foundAt == -1)
            Console.WriteLine("Value not found.");
        else
            Console.WriteLine("Found the value {0} at index {1} of the array", SEARCH_TERM, foundAt);
        Console.ReadLine( );

    }
}
