using System;
using System.Collections.Generic;
using System.ComponentModel;
using System.Data;
using System.Drawing;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Windows.Forms;
using System.IO;

namespace Proj_11_MD_V1._0
{
    public partial class Form1 : Form
    {
        EmployeeClass[] employees = new EmployeeClass[0];
        int pointer = 0;

        public Form1()
        {
            InitializeComponent();
        }

        // The Exit Menu Item event handler
        // Purpose: To close the window and terminate the application
        // Parameters: The sending object and event arguments
        // Returns: none
        private void exitToolStripMenuItem1_Click(object sender, EventArgs e)
        {
            this.Close();
        }

        // The About Menu Item event handler
        // Purpose: To tell about this application
        // Parameters: The sending object and event arguments
        // Returns: none
        private void aboutToolStripMenuItem_Click(object sender, EventArgs e)
        {
            MessageBox.Show("Michael De La Barra\nCS1400\nProj #11");
        }

        private void openToolStripMenuItem_Click(object sender, EventArgs e)
        {
            OpenFileDialog open = new OpenFileDialog();
            open.ShowDialog();

            if (open.FileName != "")
            {
                StreamReader reader = new StreamReader(open.FileName);

                bool run = true;

                while(run)
                {
                    Array.Resize(ref employees, employees.Length + 1);
                    employees[employees.Length - 1] = new EmployeeClass();
                    run = employees[employees.Length - 1].Read(ref reader);
                }

                Array.Resize(ref employees, employees.Length - 1);
                reader.Close();
            }

            pointer = 0;
            button1.Enabled = true;
            populate();
        }

        private void populate()
        {
            textBox1.Text = employees[pointer].getName();
            textBox2.Text = employees[pointer].getStreeaddress();
            textBox3.Text = employees[pointer].CalcPay().ToString("C");
            pointer++;
        }

        private void button1_Click(object sender, EventArgs e)
        {
            if (pointer < employees.Length)
            {
                populate();
            }
            else
            {
                button1.Enabled = false;
            }
        }
    }
}
