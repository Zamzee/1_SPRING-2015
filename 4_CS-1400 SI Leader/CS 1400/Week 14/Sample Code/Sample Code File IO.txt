using System;
using System.IO;

class Program
{
    static void Main()
    {

        // This line of code gets the current path to the user's My Documents Folder
        string environment = System.Environment.GetFolderPath(System.Environment.SpecialFolder.Personal) + "\\";
        Console.WriteLine(environment);  // display it for fun
        Console.WriteLine("Enter a file name in My Documents: ");
        string input = Console.ReadLine();

        // concatenate the path to the file name
        string path = environment + input;

        // now we can use the full path to get the document
        StreamReader myFile = new StreamReader(path);

        // read two values from the file and add them together
        int n1 = int.Parse(myFile.ReadLine());
        int n2 = int.Parse(myFile.ReadLine());
        int sum = n1 + n2;

        // Output the results
        Console.WriteLine("Read the values {0:d} and {1:d} from the file.", n1, n2);
        Console.WriteLine("The sum of these values is {0:d}", sum);

        Console.ReadLine();
    }//End Main()
}//End class Program
